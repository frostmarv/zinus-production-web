// Development origins explicitly allowed
const developmentOrigins = [
  'http://127.0.0.1:5000',     // Frontend dev server ✅
  'http://localhost:5000',      // Alternative
  `https://${REPLIT_DEV_DOMAIN}` // Replit frontend domain
];
// Smart CORS handler
app.enableCors({
  origin: (origin, callback) => {
    if (!origin) return callback(null, true); // Allow no-origin requests
    
    if (allowedOrigins.includes(origin)) {
      callback(null, true); // ✅ Allowed
    } else if (origin.includes('.replit.dev')) {
      callback(null, true); // ✅ Any replit.dev domain
    } else {
      console.warn(`⚠️ CORS blocked: ${origin}`);
      callback(new Error('Not allowed by CORS')); // ❌ Blocked
    }
  },
  credentials: true
});
